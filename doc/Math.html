def atan2(x,y)
  case
  when (Unit === x && Unit === y) && (x !~ y)
    raise ArgumentError, "Incompatible Units"
  when (Unit === x && Unit === y) && (x =~ y)
    unit_atan2(x.base_scalar, y.base_scalar)
  when (Unit === x || Unit === y)
    raise ArgumentError, "Incompatible Units"
  else
    unit_atan2(x,y)
  end
end

def hypot(x,y)
  if Unit === x && Unit === y
    (x**2 + y**2)**(1/2)
  else
    unit_hypot(x,y)
  end
end
